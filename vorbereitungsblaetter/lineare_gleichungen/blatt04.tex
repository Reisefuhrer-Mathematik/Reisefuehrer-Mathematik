\begin{theorem}
    Die Lösungsmenge einer Ungleichung bleibt unverändert, wenn man
    \begin{itemize}
        \item zu jeder Seite eine Zahl $a\in\mathbb{R}$ addiert.
        \item jede Seite mit einer Zahl $a>0$ multipliziert.
    \end{itemize}
\end{theorem}

\subsection*{Aufgabenteil}
