\begin{exercise}{difficult}
    Die Funktion $f$ sei stückweise über
    \[f=\begin{cases}
        -\frac{1}{2}x+1 & \text{für}~x<0\\
        4x+1 & \text{für}~0\leq x<1\\
        6-x & \text{für}~1\leq x<5\\
        3x-14 & \text{für}~5\leq x
    \end{cases}\]
    definiert. Bestimme alle Nullstellen von $f$.
\end{exercise}
